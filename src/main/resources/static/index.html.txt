    <script>
        function predictDisease() {
            const fileInput = document.getElementById('imageInput');
            const file = fileInput.files[0];
            
            if (!file) {
                alert('Please select an image file first');
                return;
            }
            
            // Show mock results for demonstration
            displayMockResults();
        }
        
        function displayMockResults() {
            const resultDiv = document.getElementById('result');
            const predictionsDiv = document.getElementById('predictions');
            
            // Mock data - in real application, this would come from your Java backend
            const mockPredictions = [
                { 
                    className: "early_blight", 
                    confidence: 0.75, 
                    description: "Fungal disease causing concentric rings on leaves", 
                    treatment: "Apply fungicides containing chlorothalonil" 
                },
                { 
                    className: "late_blight", 
                    confidence: 0.20, 
                    description: "Destructive fungal disease with water-soaked lesions", 
                    treatment: "Use copper-based fungicides and remove infected plants" 
                },
                { 
                    className: "healthy", 
                    confidence: 0.05, 
                    description: "Plant shows no signs of disease", 
                    treatment: "Continue current care practices" 
                }
            ];
            
            let html = '';
            mockPredictions.forEach(pred => {
                const confidencePercent = (pred.confidence * 100).toFixed(1);
                html += `
                <div class="prediction-item">
                    <strong>${pred.className.replace('_', ' ').toUpperCase()}</strong>: ${confidencePercent}% confidence<br>
                    <small>${pred.description}</small><br>
                    <em style="color: #d9534f;">üí° Treatment: ${pred.treatment}</em>
                </div>`;
            });
            
            predictionsDiv.innerHTML = html;
            resultDiv.style.display = 'block';
            
            // Scroll to results
            resultDiv.scrollIntoView({ behavior: 'smooth' });
        }
